# 数据炼金术师 - Excel实训游戏平台技术方案

## 1. 项目概述

### 1.1 项目目标
开发一款名为"数据炼金术师"的Excel实训游戏平台，基于Univer开源电子表格引擎，为"商务数据分析与应用基础"课程提供沉浸式、游戏化的实践环境，帮助学生掌握从数据采集、清洗、处理、建模到可视化的全链路技能。

### 1.2 核心价值
- **真实环境**：基于Univer引擎，提供与Excel 2016高度一致的交互体验和计算能力
- **游戏化学习**：通过剧情、经验值、成就等元素激发学习动力
- **刻意练习**：将复杂技能拆解为可循序渐进的关卡，提供即时反馈
- **全链路覆盖**：覆盖数据处理全流程的技能点
- **免费开源**：基于开源组件，部署于Cloudflare免费计划，无授权费用

### 1.3 目标用户
- 高等院校商务数据分析相关课程的学生
- 企业员工Excel技能培训
- 希望提升数据分析能力的自学者

import { ICommandService } from '@univerjs/core';
import { SetCellValueCommand, SetRangeValuesCommand, SetFormulaCommand } from '@univerjs/sheets';

// 获取命令服务
const commandService = univer.getCommandService();

// 监听单元格值变更
commandService.onCommandExecuted((command) => {
  if (command.id === SetCellValueCommand.id) {
    const { cell, value } = command.params;
    console.log(`Cell ${cell} changed to:`, value);
    // 记录用户操作
    recordUserAction('setCellValue', { cell, value });
  }
  
  if (command.id === SetFormulaCommand.id) {
    const { cell, formula } = command.params;
    console.log(`Formula set in ${cell}:`, formula);
    // 记录用户操作
    recordUserAction('setFormula', { cell, formula });
  }
  
  if (command.id === SetRangeValuesCommand.id) {
    const { range, values } = command.params;
    console.log(`Range ${range} changed to:`, values);
    // 记录用户操作
    recordUserAction('setRangeValues', { range, values });
  }
});

// 记录用户操作序列
const userActions = [];
function recordUserAction(actionType, details) {
  userActions.push({
    timestamp: Date.now(),
    actionType,
    details
  });
}

### 3.4 实时校验实现

**公式指纹解析**：

```javascript
function parseFormulaFingerprint(formula) {
  // 提取公式中的函数名
  const functionRegex = /\b([A-Za-z]+)\s*\(/g;
  const functions = [];
  let match;
  
  while ((match = functionRegex.exec(formula)) !== null) {
    functions.push(match[1].toUpperCase());
  }
  
  // 检查是否使用了绝对引用
  const hasAbsoluteReference = /\$[A-Z]+\$\d+/.test(formula);
  
  return {
    functions,
    hasAbsoluteReference
  };
}

// 校验函数
async function validateTask(taskData) {
  const workbook = univer.getWorkbook();
  const worksheet = workbook.getActiveSheet();
  
  let allCorrect = true;
  const validationResults = [];
  
  // 遍历目标单元格
  for (const cell of taskData.targetCells) {
    const validationRule = taskData.validationRules[cell];
    const userValue = worksheet.getCellValue(cell);
    const userFormula = worksheet.getCellFormula(cell);
    
    let cellCorrect = false;
    let errorMessage = '';
    
    // 验证值
    if (validationRule.type === 'value') {
      if (userValue === validationRule.expectedValue) {
        cellCorrect = true;
      } else {
        errorMessage = `值不正确，预期值为 ${validationRule.expectedValue}`;
      }
    }
    
    // 验证公式指纹
    if (validationRule.formulaFingerprint && userFormula) {
      const fingerprint = parseFormulaFingerprint(userFormula);
      const hasRequiredFunctions = validationRule.formulaFingerprint.every(func => 
        fingerprint.functions.includes(func)
      );
      
      if (!hasRequiredFunctions) {
        cellCorrect = false;
        errorMessage = `请使用 ${validationRule.formulaFingerprint.join(', ')} 函数`;
      }
    }
    
    validationResults.push({
      cell,
      correct: cellCorrect,
      errorMessage
    });
    
    if (!cellCorrect) {
      allCorrect = false;
    }
  }
  
  // 验证操作路径
  if (taskData.expectedActionSequence) {
    const actionTypes = userActions.map(action => action.actionType);
    const sequenceMatch = taskData.expectedActionSequence.every((action, index) => 
      actionTypes[index] === action
    );
    
    if (!sequenceMatch) {
      allCorrect = false;
      validationResults.push({
        type: 'sequence',
        correct: false,
        errorMessage: '操作顺序不正确，请按照提示步骤操作'
      });
    }
  }
  
  return {
    success: allCorrect,
    results: validationResults,
    userActions
  };
}
```

### 4.4 自适应提示机制

```javascript
// 错误记录与自适应提示
const errorTracking = {
  errors: {},
  errorCounts: {}
};

function recordError(taskId, errorType) {
  if (!errorTracking.errors[taskId]) {
    errorTracking.errors[taskId] = [];
    errorTracking.errorCounts[taskId] = 0;
  }
  
  errorTracking.errors[taskId].push({
    timestamp: Date.now(),
    errorType
  });
  
  errorTracking.errorCounts[taskId]++;
  
  // 当错误次数≥3时，显示简化版讲解或降低难度
  if (errorTracking.errorCounts[taskId] >= 3) {
    showAdaptiveHint(taskId, errorType);
  }
}

function showAdaptiveHint(taskId, errorType) {
  // 根据错误类型显示相应的提示
  switch (errorType) {
    case 'formulaSyntax':
      showVideoTutorial('formula-basics');
      break;
    case 'logicError':
      showStepByStepGuide(taskId);
      break;
    case 'functionError':
      showFunctionReference(getRequiredFunction(taskId));
      break;
  }
  
  // 降低难度（如显示更详细的步骤提示）
  lowerTaskDifficulty(taskId);
}
```
